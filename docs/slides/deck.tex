% ================================================================ 2  BACKGROUND
\begin{frame}{Expressive performance}
  The score specifies pitches and note values. Performers add the timing,
  articulation, and velocity of every note. These deviations are the expression.
  \vspace{-2mm}
  \figslotw{deck_concept.png}{0.73}{the score, the same bars as a piano-roll
    (written vs played), and the three per-note deviation channels}
\end{frame}

% ================================================================ 3  PROBLEM
\begin{frame}{Problem: infer how the music was played, with reliable uncertainty}
  The goal: a probabilistic model of \textbf{every per-note expressive
  parameter a performer controls}: timing, articulation, velocity, and, on
  continuously pitched instruments, intonation and vibrato, ultimately
  inferred from audio alone. Every estimate must carry an error bar that can
  be trusted note by note.
  \vspace{2mm}

  The first completed stage: piano, from aligned performance MIDI,
  \begin{equation*}
    y_i \;=\; \bigl[\;\underbrace{\tau_i}_{\text{onset timing}},\;\;
      \underbrace{\log r_i}_{\text{articulation}},\;\;
      \underbrace{v_i}_{\text{velocity}}\;\bigr] \in \R^{3}
    \qquad\text{per written note.}
  \end{equation*}
  \begin{itemize}
    \item Not full transcription: the score is known.
    \item The output is a \textbf{per-note posterior}: an estimate and a
          calibrated error bar.
  \end{itemize}
\end{frame}

% ================================================================ 4  APPROACH
\begin{frame}{Approach: one prior, three kinds of observation}
  One \textbf{Gaussian process (GP)} prior over per-note expressive fields, on
  a graph built from the score. Only the observation changes: per-note MIDI
  values (Phase 1), noisy $f_0$-derived targets (Phase 2), or the waveform
  itself (Phase 3). A Phase-0 music model supplies input features to all.
  \figslotw{deck_programme.png}{0.9}{the shared prior with three swappable
    likelihood blocks (MIDI / $f_0$ / waveform)}
\end{frame}

% ================================================================ 5  RELATED WORK
\begin{frame}{Related work}
  \begin{columns}[T,onlytextwidth]
  \column{0.50\textwidth}
  \textbf{Expressive performance modelling}
  {\small
  \begin{itemize}
    \item rule systems: KTH performance rules (Friberg et al.\ 2006)
    \item linear basis-function models (Grachten \& Widmer 2012)
    \item deep rendering models: VirtuosoNet (Jeong et al.\ 2019),
          Oore et al.\ 2018
    \item review: Cancino-Chac\'on et al.\ 2018
  \end{itemize}
  All output one rendering or corpus statistics, none a calibrated
  per-note posterior.}
  \column{0.48\textwidth}
  \textbf{Gaussian processes and graphs}
  {\small
  \begin{itemize}
    \item \textbf{Gaussian Processes over Graphs} (Venkitaraman et al.\
          2018): outputs on graph nodes, Kronecker-coupled. The main
          foundation here
    \item graph Mat\'ern and heat kernels (Borovitskiy et al.\ 2021)
    \item multi-output coupling via ICM (Bonilla et al.\ 2008)
  \end{itemize}}
  \end{columns}
  \vspace{3mm}
  This work combines the two: a graph GP over score positions, with per-note
  side information, for calibrated performance inference.
\end{frame}

% ================================================================ 6  METHOD 1
\begin{frame}{Method (1): the score graph}
  Nodes are the $N$ written notes: pitch $p_i$, beat onset $b_i$, beat
  duration $d_i$. Edges connect notes close in score time and pitch
  (neighbours should be played similarly):
  \vspace{-1mm}
  \begin{equation*}
    W_{ij} \;=\; \exp\!\Bigl( -\tfrac{(b_i - b_j)^2}{2\,\ell_b^2}
                             \;-\; \tfrac{(p_i - p_j)^2}{2\,\ell_p^2} \Bigr),
    \qquad
    \Lg \;=\; D - W .
  \end{equation*}
  \vspace{-2mm}
  {\centering\small $\ell_b, \ell_p$: length-scales. $D_{ii} = \sum_j W_{ij}$.
  $\Lg$: the graph Laplacian.\par}
  \figslotw{deck_scoregraph.png}{0.80}{a few bars of real score as nodes,
    edges drawn by time / pitch / chord relation}
\end{frame}

% ================================================================ 7  METHOD 2
\begin{frame}{Method (2): a GP on the graph, channels coupled}
  Eigendecompose the Laplacian, $\Lg = U \diag(\nu)\, U^{\!\top}$, and shape its
  spectrum with one parameter $s$:
  \begin{equation*}
    K_G(s) \;=\; U\, g(\nu; s)\, U^{\!\top},
    \qquad
    g(\nu; s) \;=\; \frac{1}{1 + s\,\nu}
    \quad\text{(graph Mat\'ern, } \alpha = 1\text{)} .
  \end{equation*}
  Smooth-on-the-graph components keep their variance, rough ones are shrunk.
  \vspace{1mm}

  Timing, articulation, and velocity are correlated, so the three channels
  share the graph through a learned $3 \times 3$ matrix $B$
  ($c, c'$ index the channels):
  \begin{equation*}
    \operatorname{Cov}\bigl[\, y_{i,c},\; y_{j,c'} \bigr]
      \;=\; B_{cc'}\, \bigl[K_G(s)\bigr]_{ij}
    \qquad\Longleftrightarrow\qquad
    \operatorname{Cov}[\,y\,] \;=\; B \otimes K_G(s).
  \end{equation*}
  An observed channel at one note then informs the other channels at
  neighbouring notes.
\end{frame}

% ================================================================ 8  METHOD 3
\begin{frame}{Method (3): side information as kernels}
  Any per-note side information enters the same way. In Phase 1 there are
  two sets: simple score descriptors $X_{\mathrm{feat}}$ (register,
  duration, meter, position) and per-note features $H$ from a small music
  model trained on MAESTRO. Designing richer score features is future work.
  \vspace{1mm}

  A linear mean with unknown weights $\beta$, integrated out, becomes a
  kernel term ($X$ = either feature matrix, $f_G$ = the graph GP of
  Method (2), $c_X$ = the prior scale of set $X$'s weights):
  \begin{equation*}
    y \;=\; X\beta + f_G, \quad \beta \sim \N(0,\, c_X\,I)
    \qquad\Longrightarrow\qquad
    \operatorname{Cov}[\,y\,] \;=\; c_X\, X X^{\!\top} + \operatorname{Cov}[\,f_G\,].
  \end{equation*}
  \begin{itemize}
    \item So the regression weights are not fixed once across the corpus: they
          are \textbf{re-inferred for every piece}, jointly with everything
          else, and their uncertainty enters the posterior.
    \item This per-piece adaptation turns out to be the main source of the
          accuracy gain (results).
  \end{itemize}
\end{frame}


% ================================================================ 10  METHOD 4
\begin{frame}{Method (4): the covariance of one performance}
  Stack all three channels of the $N$ notes into one vector $y \in \R^{3N}$.
  Every component becomes one term of its covariance $K$:
  \begin{equation*}
    K \;=\;
    \underbrace{\textcolor{okblue}{B \otimes K_G(s)}}_{\text{coupled graph GP}}
    \;+\; \underbrace{\textcolor{okamber}{\textstyle\sum_f \diag(c_f)\otimes X_f X_f^{\!\top}}}_{\text{side information}}
    \;+\; \underbrace{\textcolor{okverm}{\diag(\varsigma^2)\otimes I}}_{\text{noise}}
  \end{equation*}
  {\small $f$ runs over the two feature sets, giving per-channel scales
  $c_{\mathrm{feat}}$ and $c_{\mathrm{mm}}$. $\varsigma^2$ = the three
  per-channel noise variances.}
  \vspace{1mm}

  All hyperparameters $\theta = (s, B, \{c_f\}, \varsigma^2)$ from one exact
  evidence per piece. Prediction is conjugate:
  \begin{equation*}
    \log p(\tilde y_o \mid \theta) = \log \N\!\bigl(\tilde y_o;\, 0,\, K_{oo}\bigr),
    \qquad
    m = C_{\cdot o} K_{oo}^{-1} \tilde y_o, \quad
    \Sigma_y = C - C_{\cdot o} K_{oo}^{-1} C_{o\cdot},
  \end{equation*}
  {\small $\tilde y_o$: the observed values. $o$: the observed entries of
  $y$. $C = K - \diag(\varsigma^2)\otimes I$: the noise-free part. $m,
  \Sigma_y$: posterior mean and covariance. No variational approximation
  anywhere in Phase 1, and the baselines are exact special cases (diagonal
  $B$, zero $c_f$, fixed cross-piece mean).}
\end{frame}

% ================================================================ 11  SETUP
\begin{frame}{Experiment setup}
  \begin{itemize}
    \item \textbf{Data}: ASAP piano (aligned score + performance MIDI).
          Performances that appeared in pretraining are removed
          ($1036 \to 653$).
    \item \textbf{Task}: hide notes, predict them from the visible rest,
          at \textbf{six masking levels}: 50, 40, 30, 20, 10\% hidden, and
          leave-one-out.
    \item \textbf{Metrics}, per channel and per level: RMSE for accuracy,
          NLL and coverage for the error bars. Comparisons are paired per
          piece.
  \end{itemize}
\end{frame}

          gains sit in articulation and velocity.
    \item In the full model (music model present) the added edge families
          become redundant.
  \end{itemize}}
\end{frame}




% ================================================================ 22  FUTURE: PHASE 2
\begin{frame}{Future work: Phase 2, intonation and vibrato}
  Continuously pitched instruments. The channels grow: intonation offset
  $c_i$ in cents, and vibrato rate / depth / phase $f_i^{\mathrm{vib}},
  \gamma_i, \delta_i^{\mathrm{vib}}$ in the cents curve $c_i(t)$:
  \begin{equation*}
    y_i = [\,\tau_i, \log r_i, c_i, f_i^{\mathrm{vib}}, \gamma_i,
    \delta_i^{\mathrm{vib}}\,], \qquad
    c_i(t) = c_i + \gamma_i \sin\bigl(2\pi f_i^{\mathrm{vib}}
    (t - \delta_i^{\mathrm{vib}})\bigr) + \epsilon .
  \end{equation*}
  \textbf{Pipeline}: $f_0$ tracking $\to$ cents curve $\to$ per-note joint
  least-squares fit (with parameter variances) $\to$ the same GP, now with
  per-(note, channel) masks and a noise level that varies per note.
  \vspace{-2mm}
  {\small
  \begin{itemize}
    \item Notes too short for a vibrato fit would have missing cells, and
          the posterior would fill them from the graph.
    \item Planned data: URMP (per-instrument audio + scores), then solo
          vocal corpora.
    \item Planned evaluation: the same held-out imputation, with a fresh
          preregistered test set before any claim.
    \item Main risk: audio-to-score alignment error enters $\tau$ directly and
          is correlated along time. Mitigation is to model it in the noise
          or exclude $\tau$ from Phase-2 claims.
  \end{itemize}}
\end{frame}


% ================================================================ 24  FUTURE: PHASE 3
\begin{frame}{Future work: Phase 3, waveform likelihood}
  Replace per-note targets with the recorded audio itself:
  \begin{equation*}
    x \;=\; \Phi(z)\, a + \epsilon, \qquad
    p(x \mid z) \;=\; \N\!\bigl(x;\; \Phi\mu_a,\;
    \Phi\Sigma_a\Phi^{\!\top} + K_n\bigr),
  \end{equation*}
  where $z$ are the per-note expressive fields (under the same graph prior),
  $\Phi(z)$ is a differentiable harmonic synthesis basis, and the amplitudes
  $a \sim \N(\mu_a, \Sigma_a)$ integrate out in closed form
  ($K_n$: audio-noise covariance).
  \begin{itemize}
    \item The amplitude step is conjugate, so it stays exact. The open part
          is inference over $z$: with no per-note observations conjugacy is
          lost, and the plan is Laplace or variational inference, initialised
          from the score.
    \item Planned evaluation: synthetic audio first (parameter recovery,
          identifiability), then monophonic recordings, with
          posterior-predictive resynthesis as the end-to-end check.
    \item Open question: how to measure calibration without per-note ground
          truth.
  \end{itemize}
\end{frame}

% ================================================================ 25  SUMMARY
\begin{frame}{Summary}
  \begin{itemize}
    \item One GP prior on the score graph covers the whole program. Only
          the observation model changes between phases.
    \item Phase 1: best accuracy in every channel, at every masking level
          from 50\% hidden to leave-one-out, on held-out validation pieces.
          The graph carries the calibration.
    \item Where the gains come from is measured, not assumed: per-piece
          feature weighting for accuracy, the graph for calibration.
    \item Next: richer score features as a Phase-1 extension, then
          Phases 2 and 3 with the same prior and the observation model
          swapped for noisy estimates, then for the waveform.
  \end{itemize}
\end{frame}
