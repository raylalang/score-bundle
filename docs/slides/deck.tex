% ================================================================ 2  BACKGROUND
\begin{frame}{Expressive performance}
  The score specifies pitches and note values. Performers add everything else:
  \begin{itemize}
    \item timing: notes pushed and pulled around the beat;
    \item articulation: held longer or shorter than written;
    \item dynamics: shaped louder and softer across phrases.
  \end{itemize}
  \vspace{1mm}
  These deviations are the expression. Modelling them matters for performance
  analysis, pedagogy, and expressive rendering.
  \figslot{deck_concept.png}{2.6cm}{one bar as a piano-roll, with the played
    timing / duration / loudness deviating from the written values}
\end{frame}

% ================================================================ 3  PROBLEM
\begin{frame}{Problem: infer the deviations, with reliable uncertainty}
  Given the score and a performance, infer for every written note
  \begin{equation*}
    y_i \;=\; \bigl[\;\underbrace{\tau_i}_{\text{onset timing}},\;\;
      \underbrace{\log r_i}_{\text{articulation}},\;\;
      \underbrace{v_i}_{\text{dynamics}}\;\bigr] \in \R^{3}.
  \end{equation*}
  \begin{itemize}
    \item Not full transcription: the score is known.
    \item The target output is a \textbf{per-note posterior}: an estimate
          \emph{and} an error bar that can be trusted note by note.
    \item Most prior work predicts a rendering; per-note \emph{calibrated
          uncertainty} is the gap this work addresses.
  \end{itemize}
\end{frame}

% ================================================================ 4  APPROACH OVERVIEW
\begin{frame}{Approach: one prior, three kinds of observation}
  One GP prior over per-note expressive fields on a graph built from the
  score. Only the observation model changes across phases:
  \begin{center}
  {\footnotesize
  \begin{tabular}{@{}llll@{}}
  \toprule
  Phase & Observation & Channels & Status \\
  \midrule
  0 \; music model & MIDI corpora (MAESTRO) & --- & done; feature provider \\
  1 \; piano & performance MIDI, per note & $\tau, \log r, v$ & \textbf{done, confirmed} \\
  2 \; mono.\ pitched & $f_0$-derived targets, noisy & $+\,c$, vibrato & machinery built \\
  3 \; waveform & the audio itself & all latent & scoped \\
  \bottomrule
  \end{tabular}}
  \end{center}
  \vspace{-1mm}
  \figslot{deck_programme.png}{2.4cm}{the shared prior with three swappable
    likelihood blocks (MIDI / $f_0$ / waveform)}
\end{frame}

% ================================================================ 5  RELATED WORK
\begin{frame}{Related work}
  \begin{columns}[T,onlytextwidth]
  \column{0.50\textwidth}
  \textbf{Expressive performance modelling}
  \begin{itemize}
    \item rule systems; sequence models trained on aligned corpora
          (Cancino-Chac\'on et al.; Oore et al.)
    \item typically predicts one rendering or aggregate statistics
  \end{itemize}
  \column{0.48\textwidth}
  \textbf{Gaussian processes on graphs}
  \begin{itemize}
    \item graph Mat\'ern and heat kernels (Borovitskiy et al.\ 2021)
    \item multi-output coupling via ICM (Bonilla et al.\ 2008)
  \end{itemize}
  \end{columns}
  \vspace{3mm}
  This work combines the two: standard GP components, applied to per-note
  performance inference. Deviations from the textbook forms exist only where
  forced (identifiability, numerical floors).
\end{frame}

% ================================================================ 6  METHOD: MODEL
\begin{frame}{Method: the model}
  \begin{equation*}
    K \;=\;
    \underbrace{\textcolor{okblue}{B \otimes K_G(s)}}_{\substack{\text{graph GP, channels}\\ \text{coupled by } B}}
    \;+\; \underbrace{\textcolor{okamber}{\textstyle\sum_f \diag(c_f)\otimes X_f X_f^{\!\top}}}_{\substack{\text{score features and music-model}\\ \text{embeddings, as linear kernels}}}
    \;+\; \underbrace{\textcolor{okverm}{\diag(\varsigma^2)\otimes I}}_{\text{noise}}
  \end{equation*}
  \begin{itemize}
    \item Graph: notes are connected by proximity in score time, pitch, and
          voice; $K_G$ is a spectral kernel of its Laplacian.
    \item Side information enters as linear kernels: equivalent to a Bayesian
          linear mean whose weights are re-inferred \emph{for each piece}.
    \item All hyperparameters fit by the exact per-piece marginal likelihood;
          the posterior is closed-form.
  \end{itemize}
  \vspace{-2mm}
  \figslot{arch_phase1_deck.png}{2.1cm}{architecture: graph branch (blue),
    feature branch (amber), likelihood (vermillion)}
\end{frame}

% ================================================================ 7  METHOD: EVALUATION
\begin{frame}{Method: evaluation setup \hfill \DEV}
  \begin{itemize}
    \item Data: ASAP piano (aligned score + performance MIDI), filtered so no
          test performance appeared in pretraining ($1036 \to 653$).
    \item Task: hide a fraction of notes, predict them from the visible rest.
          Results are unchanged from 50\% hidden down to a single hidden note.
    \item Metrics: RMSE, plus NLL / coverage / PIT for the error bars.
          A method must improve both to count as better.
    \item Comparisons are paired per piece, with bootstrap CIs over pieces.
  \end{itemize}
  \vspace{1mm}
  \textbf{Baselines}: independent notes, temporal smoothing, feature and
  embedding regressors, and a \textbf{two-stage baseline}: fit a mean
  predictor across pieces, then smooth its residuals with the graph, one
  channel at a time. Each is a special case of the proposed model.
\end{frame}

% ================================================================ 8  METHOD: HYGIENE
\begin{frame}{Method: guarding the numbers}
  \begin{itemize}
    \item \textbf{Leakage}: the music model's read-out is provably blind to
          each note's own target; the pipeline is audited end-to-end.
    \item \textbf{Selection}: 30 pieces were used for all development.
          The final claims were then fixed in advance and evaluated
          \textbf{exactly once} on 20 untouched pieces.
    \item The one claim that failed on that test is reported.
  \end{itemize}
  \vspace{4mm}

% ================================================================ 14  RESULTS: FINDINGS
\begin{frame}{Findings: what predicts expressive deviations \hfill \DEV}
  Three measurements agree about tonality:
  \begin{enumerate}
    \item a circle-of-fifths pitch metric in the graph makes results
          \emph{worse} ($+0.009^{*}$);
    \item explicit tonal features (key, scale degree) add nothing
          ($-0.002$ ns);
    \item probes find no tonal information in the music model's embeddings.
          They do find bass membership (AUC $0.99$), meter, phrase, register.
  \end{enumerate}
  \vspace{1mm}
  Tonal structure does not predict piano timing, articulation, or dynamics.
  Rhythm, voicing, and register do. Chord and voice-leading \emph{edges} help
  ($-0.008^{*}$); the choice of spectral filter does not (backup slide).
\end{frame}

% ================================================================ 16  FUTURE WORK
\begin{frame}{Future work: Phases 2 and 3 \hfill \SYNTH}
  \textbf{Phase 2} (intonation and vibrato, continuously pitched instruments):
  \begin{itemize}
    \item targets become noisy estimates, and can be missing per (note,
          channel); the posterior now handles this exactly, in closed form;
    \item synthetic pilot with known ground truth: the graph recovers vibrato
          parameters even for notes too short to estimate directly;
    \item open: real audio, $f_0$ tracking, alignment error.
  \end{itemize}
  \vspace{1mm}
  \textbf{Phase 3} (waveform likelihood $x = \Phi(z)a + \epsilon$):
  \begin{itemize}
    \item amplitudes marginalise exactly (implemented, tested); inference over
          $z$ is the open step; conjugacy is lost there.
  \end{itemize}
  \vspace{-1mm}
  \figslot{arch_phase2_deck.png}{1.8cm}{Phase-2 architecture: only the
    likelihood block changes (exists)}
\end{frame}

% ================================================================ 17  SUMMARY
\begin{frame}{Summary}
  \begin{itemize}
    \item One GP prior on the score graph covers the whole program; only the
          observation model changes between phases.
    \item Phase 1: accuracy gain confirmed on a preregistered one-shot test;
          the graph's calibration role confirmed; replicated on fresh pieces.
    \item Where the gains come from is measured, not assumed: per-piece
          feature weighting for accuracy, the graph for calibration.
    \item Negative results are kept: tonality does not predict these
          deviations; whole-piece summaries do not benefit.
    \item Phase 2 machinery is built and validated on synthetic ground truth.
  \end{itemize}
\end{frame}

\begin{frame}{Backup: statistical robustness \hfill \DEV}
  \begin{itemize}
    \item Main contrast per channel: $\tau\ -0.009^{*}$,
          $\log r\ -0.036^{*}$, $v\ -0.006^{*}$. Variance-standardized pooling
          strengthens the result (27/30 pieces).
    \item Percentile, basic, and BCa bootstrap, Wilcoxon signed-rank, and
          exact sign tests agree on every key contrast.
    \item All load-bearing starred contrasts survive Benjamini--Hochberg at
          $q = 0.05$ over 19 development contrasts.
    \item Composer-clustered bootstrap keeps every key contrast significant.
          12 mask seeds move pooled numbers by at most 0.003 RMSE.
          Piece-level coverage never falls below 0.897.
  \end{itemize}

\begin{frame}{Backup: the Gaussian tail and the Student-$t$ fix \hfill \DEV}
  \begin{itemize}
    \item Failure mode (one test piece and one dev replica): a very steady
          piece gives a tiny fitted timing scale; a few outlier notes then
          explode the pooled NLL while RMSE and coverage stay normal
          (dev replica: NLL $+34.7$, coverage 0.90).
    \item Student-$t$ timing likelihood (EM prototype, unit-tested): on the dev
          tail cell, NLL goes from $+34.7$ to $\bm{-0.96}$ under its own
          predictive. RMSE and coverage are unchanged.
    \item On tail-free pieces it costs a small premium.
    \item It is applied to no thesis number: the test set is spent, so the fix
          waits for its own preregistered confirmation.
  \end{itemize}
\end{frame}
